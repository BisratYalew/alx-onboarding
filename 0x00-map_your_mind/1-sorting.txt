INPUT variable num3;
Declare variable temp;

    Initialise temp = num1
    num1 = num2

    Initialize temp = num1
    num1 = num3
    num3 = temp
IF num2 > num3
    Initialize temp = num2
    num2 = num3
    num3 = temp
